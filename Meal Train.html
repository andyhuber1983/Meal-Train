  <script>

    // ====================== PASTE YOUR FIREBASE CONFIG HERE ======================

    const firebaseConfig = {

      apiKey: "YOUR_API_KEY",

      authDomain: "your-project.firebaseapp.com",

      projectId: "your-project",

      storageBucket: "your-project.appspot.com",

      messagingSenderId: "123456789",

      appId: "1:123456789:web:abcdef123456"

    };

    // =============================================================================


    if (firebaseConfig.apiKey === "YOUR_API_KEY") {

      alert("⚠️ Please replace the firebaseConfig object at the top of the <script> with your own from Firebase console!");

    }


    firebase.initializeApp(firebaseConfig);

    const db = firebase.firestore();

    const collectionRef = db.collection("mealDays");


    // Default schedule (edit dates or days here if needed)

    const defaultSlots = [

      { day: "Monday",    date: "March 16, 2026", assigned: "", meal: "" },

      { day: "Tuesday",   date: "March 17, 2026", assigned: "", meal: "" },

      { day: "Wednesday", date: "March 18, 2026", assigned: "", meal: "" },

      { day: "Thursday",  date: "March 19, 2026", assigned: "", meal: "" },

      { day: "Friday",    date: "March 20, 2026", assigned: "", meal: "" }

    ];


    let schedule = [];


    // Auto-seed if collection is empty

    async function seedIfNeeded() {

      try {

        const snap = await collectionRef.get();

        if (snap.empty) {

          const batch = db.batch();

          defaultSlots.forEach(slot => {

            const docRef = collectionRef.doc(slot.day);

            batch.set(docRef, slot);

          });

          await batch.commit();

          console.log("✅ Initial schedule seeded in Firebase");

        }

      } catch (e) {

        console.error("Seed error:", e);

      }

    }


    // Real-time listener

    function startRealtimeListener() {

      collectionRef.onSnapshot((snapshot) => {

        schedule = [];

        snapshot.forEach(doc => {

          const data = doc.data();

          schedule.push({ ...data, docId: doc.id });

        });


        // Keep fixed order

        const dayOrder = ["Monday","Tuesday","Wednesday","Thursday","Friday"];

        schedule.sort((a, b) => dayOrder.indexOf(a.day) - dayOrder.indexOf(b.day));


        renderTable();

        populateDaySelect();

      });

    }


    function renderTable() {

      const tbody = document.getElementById('tableBody');

      tbody.innerHTML = '';


      schedule.forEach(slot => {

        const row = document.createElement('tr');

        row.className = slot.assigned ? 'taken' : 'available';

        

        row.innerHTML = `

          <td style="font-weight:600;">${slot.day}</td>

          <td style="color:var(--zinc-400);">${slot.date}</td>

          <td>

            ${slot.assigned 

              ? `<span style="color:var(--emerald-400); font-weight:600;">${slot.assigned}</span>` 

              : `<span class="available-badge">✅ Available</span>`}

          </td>

          <td>

            ${slot.meal 

              ? `<span class="meal-text">${slot.meal}</span>` 

              : `<span class="empty-meal">—</span>`}

          </td>

        `;

        tbody.appendChild(row);

      });

    }


    function populateDaySelect() {

      const select = document.getElementById('daySelect');

      select.innerHTML = '<option value="">Choose an available day...</option>';

      

      let hasOpen = false;

      schedule.forEach((slot, index) => {

        if (!slot.assigned) {

          hasOpen = true;

          const option = document.createElement('option');

          option.value = index;

          option.textContent = `${slot.day} (${slot.date})`;

          select.appendChild(option);

        }

      });


      if (!hasOpen) {

        const option = document.createElement('option');

        option.textContent = "All days are already claimed ❤️";

        select.appendChild(option);

        select.disabled = true;

      } else {

        select.disabled = false;

      }

    }


    // Form submit (with race-condition protection)

    document.getElementById('mealForm').addEventListener('submit', async function(e) {

      e.preventDefault();

      

      const dayIndex = parseInt(document.getElementById('daySelect').value);

      if (isNaN(dayIndex)) return;


      const selectedSlot = schedule[dayIndex];

      if (selectedSlot.assigned) {

        alert("Sorry, that day was just claimed by someone else!");

        return;

      }


      const name = document.getElementById('name').value.trim();

      const email = document.getElementById('email').value.trim();

      const meal = document.getElementById('meal').value.trim();


      const docRef = collectionRef.doc(selectedSlot.docId || selectedSlot.day);


      try {

        await docRef.update({

          assigned: name + (email ? ` (${email})` : ''),

          meal: meal

        });


        // Show success

        document.getElementById('mealForm').style.display = 'none';

        document.getElementById('successMessage').style.display = 'block';


        setTimeout(() => {

          document.getElementById('mealForm').reset();

          document.getElementById('mealForm').style.display = 'block';

          document.getElementById('successMessage').style.display = 'none';

        }, 4000);

      } catch (err) {

        alert("Error saving: " + err.message);

      }

    });


    // Reset (for testing / new week)

    window.resetSchedule = async function() {

      if (!confirm("Clear the entire schedule in Firebase? This cannot be undone.")) return;

      

      try {

        const snap = await collectionRef.get();

        const batch = db.batch();

        snap.forEach(doc => batch.delete(doc.ref));

        await batch.commit();

        await seedIfNeeded(); // re-seed fresh dates

        alert("Schedule cleared and reset!");

      } catch (e) {

        alert("Error resetting: " + e.message);

      }

    };


    // Start everything

    (async () => {

      await seedIfNeeded();

      startRealtimeListener();

    })();

  </script>
